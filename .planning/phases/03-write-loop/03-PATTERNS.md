```typescript
  .select({
```

**Row aggregation pattern** (lines 45–60):
```typescript

const map = new Map<number, EntryListItem & { body: string; createdAt: Date }>();
for (const row of rows) {
  if (!map.has(row.id)) {
    map.set(row.id, {
      id:        row.id,
      title:     row.title,
      body:      row.body,
      createdAt: row.createdAt,
      snippet:   generateSnippet(row.body),
      dateLabel: formatEntryDate(row.createdAt, now),
      tags:      [],
    });
  }
  if (row.tagName) map.get(row.id)!.tags.push(row.tagName);
}
```

**`revalidatePath` pattern for mutations:**
```typescript
revalidatePath('/');  // Always call after mutations
revalidatePath(`/entries/${id}`);  // Revalidate specific entry page too
```

**Phase 3 additions — INSERT pattern (createEntry):**
- Drizzle `.insert().values().returning().all()` returns `[{ id: number }]` (use `[0]`)
- Return `{ id: number }` to Client Component for redirect
- Call `revalidatePath('/')` after

**Phase 3 additions — UPDATE pattern (updateEntry):**
- Drizzle `.update().set().where().run()` — `.run()` is synchronous with better-sqlite3
- Always set `updatedAt: new Date()` on content updates
- Call `revalidatePath('/')` and `revalidatePath('/entries/${id}')` after

**Phase 3 additions — DELETE pattern (deleteEntry):**
- Drizzle `.delete().where().run()`
- No manual junction cleanup — `entry_tags` has `ON DELETE CASCADE` (confirmed in db/schema.ts line 27)
- Call `revalidatePath('/')` after

**Phase 3 additions — Tag upsert pattern (setEntryTags):**
- Normalize tags server-side: `trim().toLowerCase().deduplicate`
- Delete all existing `entryTags` for the entry
- For each tag: `db.insert(tags).values({ name }).onConflictDoNothing().run()`
- Fetch the tag ID: `db.select().from(tags).where(eq(tags.name, name)).all()[0]`
- Insert `entry_tags` record

**Ref pattern for tracking creation and debounce:**

```typescript
const isSavingRef = useRef(false);
const debounceRef = useRef<ReturnType<typeof setTimeout> | null>(null);
const fadeTimerRef = useRef<ReturnType<typeof setTimeout> | null>(null);
```

**Autosave trigger function pattern:**
```typescript
const triggerAutosave = useCallback((t: string, b: string, tags: string[]) => {
  if (isSavingRef.current) return; // Skip if save in progress
  if (debounceRef.current) clearTimeout(debounceRef.current);
  
  debounceRef.current = setTimeout(async () => {
    if (!t.trim() && !b.trim()) return; // Skip if both empty
    
    isSavingRef.current = true;
    setSaveStatus('saving');
    try {
      if (entryIdRef.current === null) {
        // First save
        const { id } = await createEntry(t, b);
        entryIdRef.current = id;
        await setEntryTags(id, tags);
        router.replace(`/entries/${id}/edit`);
      } else {
        // Subsequent saves
        await updateEntry(entryIdRef.current, t, b);
        await setEntryTags(entryIdRef.current, tags);
      }
      setSaveStatus('saved');
      if (fadeTimerRef.current) clearTimeout(fadeTimerRef.current);
      fadeTimerRef.current = setTimeout(() => setSaveStatus('idle'), 2500);
    } catch {
      setSaveStatus('idle');
    } finally {
      isSavingRef.current = false;
    }
  }, 500);
}, [router]);
```

**onChange handlers pattern:**
```typescript
<input
  type="text"
  value={title}
  onChange={(e) => {
    setTitle(e.target.value);
    triggerAutosave(e.target.value, body, chipTags);
  }}
/>

<div data-color-mode="light">
  <MDEditor
    value={body}
    onChange={(val) => {
      setBody(val ?? '');
      triggerAutosave(title, val ?? '', chipTags);
    }}
    height={500}
  />
</div>
```

**Tag chip zone pattern:**
```typescript
<div className="flex flex-wrap gap-2">
  {chipTags.map((tag) => (
    <EditorTagChip
      key={tag}
      name={tag}
      onRemove={() => {
        const updated = chipTags.filter((t) => t !== tag);
        setChipTags(updated);
        triggerAutosave(title, body, updated);
      }}
    />
  ))}
  <input
    type="text"
    placeholder="Add tag (Enter or comma to create)"
    onKeyDown={(e) => {
      if (e.key === 'Enter' || e.key === ',') {
        e.preventDefault();
        const normalized = e.currentTarget.value.trim().toLowerCase();
        if (normalized && !chipTags.includes(normalized)) {
          const updated = [...chipTags, normalized];
          setChipTags(updated);
          e.currentTarget.value = '';
          triggerAutosave(title, body, updated);
        }
      }
    }}
  />
</div>
```

### `tests/actions.test.ts` (test)

**Analog:** No existing test patterns in the codebase (new for Phase 3)

**Recommended test structure** (Server Action unit tests with in-memory SQLite):
```typescript
import { describe, it, expect, beforeEach } from 'vitest';
import Database from 'better-sqlite3';
import { drizzle } from 'drizzle-orm/better-sqlite3';
import { eq } from 'drizzle-orm';
import * as schema from '@/lib/db/schema';

// For unit testing Server Actions, use an in-memory SQLite database
let testDb: ReturnType<typeof drizzle>;

beforeEach(() => {
  const sqlite = new Database(':memory:');
  
  // Initialize schema
  sqlite.exec(`
    CREATE TABLE entries (
      id INTEGER PRIMARY KEY AUTOINCREMENT,
      title TEXT NOT NULL DEFAULT '',
      body TEXT NOT NULL DEFAULT '',
      created_at INTEGER NOT NULL DEFAULT (unixepoch()),
      updated_at INTEGER NOT NULL DEFAULT (unixepoch())
    );
    CREATE TABLE tags (
      id INTEGER PRIMARY KEY AUTOINCREMENT,
      name TEXT NOT NULL UNIQUE
    );
    CREATE TABLE entry_tags (
      entry_id INTEGER NOT NULL REFERENCES entries(id) ON DELETE CASCADE,
      tag_id INTEGER NOT NULL REFERENCES tags(id) ON DELETE CASCADE,
      PRIMARY KEY (entry_id, tag_id)
    );
  `);
  
  testDb = drizzle(sqlite, { schema });
});

describe('createEntry', () => {
  it('should insert entry and return id', () => {
    const result = testDb
      .insert(schema.entries)
      .values({ title: 'Test', body: 'Content' })
      .returning({ id: schema.entries.id })
      .all();
    
    expect(result).toHaveLength(1);
    expect(result[0].id).toBe(1);
  });
});

describe('updateEntry', () => {
  it('should update entry title and body', () => {
    testDb.insert(schema.entries).values({ title: 'Old', body: 'Old' }).run();
    
    testDb
      .update(schema.entries)
      .set({ title: 'New', body: 'New Content', updatedAt: new Date() })
      .where(eq(schema.entries.id, 1))
      .run();
    
    const entry = testDb.select().from(schema.entries).where(eq(schema.entries.id, 1)).all()[0];
    expect(entry.title).toBe('New');
    expect(entry.body).toBe('New Content');
  });
});

describe('deleteEntry with cascade', () => {
  it('should delete entry and cascade to entry_tags', () => {
    // Insert entry
    testDb.insert(schema.entries).values({ title: 'Test', body: 'Content' }).run();
    
    // Insert tag and entry_tag
    testDb.insert(schema.tags).values({ name: 'tag1' }).run();
    testDb.insert(schema.entryTags).values({ entryId: 1, tagId: 1 }).run();
    
    // Delete entry
    testDb.delete(schema.entries).where(eq(schema.entries.id, 1)).run();
    
    // Verify entry is gone
    const entry = testDb.select().from(schema.entries).where(eq(schema.entries.id, 1)).all();
    expect(entry).toHaveLength(0);
    
    // Verify entry_tags cascade deleted
    const entryTags = testDb.select().from(schema.entryTags).where(eq(schema.entryTags.entryId, 1)).all();
    expect(entryTags).toHaveLength(0);
  });
});

describe('setEntryTags upsert pattern', () => {
  it('should insert new tags or skip duplicates', () => {
    testDb.insert(schema.entries).values({ title: 'Test', body: 'Content' }).run();
    testDb.insert(schema.tags).values({ name: 'existing' }).run();
    
    // Try to insert 'existing' and new 'new-tag'
    testDb.insert(schema.tags).values({ name: 'existing' }).onConflictDoNothing().run();
    testDb.insert(schema.tags).values({ name: 'new-tag' }).onConflictDoNothing().run();
    
    const tags = testDb.select().from(schema.tags).all();
    expect(tags).toHaveLength(2);
    expect(tags.map(t => t.name).sort()).toEqual(['existing', 'new-tag']);
  });
});
```

### Server Actions Pattern (All CRUD actions)

**Source:** `src/lib/actions.ts` (existing pattern — extend with Phase 3 actions)

**Pattern applies to:** `createEntry`, `updateEntry`, `deleteEntry`, `setEntryTags`

```typescript
'use server';

import { revalidatePath } from 'next/cache';

export async function createEntry(title: string, body: string): Promise<{ id: number }> {
  // Synchronous Drizzle query with better-sqlite3
  const result = db
    .insert(entries)
    .values({ title, body })
    .returning({ id: entries.id })
    .all();
  
  // Always revalidate after mutation
  revalidatePath('/');
  
  return { id: result[0].id };
}

export async function updateEntry(id: number, title: string, body: string): Promise<void> {
  db.update(entries)
    .set({ title, body, updatedAt: new Date() })
    .where(eq(entries.id, id))
    .run();
  
  // Revalidate both home and detail page
  revalidatePath('/');
  revalidatePath(`/entries/${id}`);
}

export async function deleteEntry(id: number): Promise<void> {
  db.delete(entries).where(eq(entries.id, id)).run();
  revalidatePath('/');
}

export async function setEntryTags(entryId: number, tagNames: string[]): Promise<void> {
  // Normalize server-side as guard
  const normalized = [...new Set(
    tagNames.map(t => t.trim().toLowerCase()).filter(Boolean)
  )];
  
  // Clear existing
  db.delete(entryTags).where(eq(entryTags.entryId, entryId)).run();
  
  // Upsert and link
  for (const name of normalized) {
    db.insert(tags).values({ name }).onConflictDoNothing().run();
    const tag = db.select().from(tags).where(eq(tags.name, name)).all()[0];
    db.insert(entryTags).values({ entryId, tagId: tag.id }).onConflictDoNothing().run();
  }
  
  revalidatePath('/');
  revalidatePath(`/entries/${entryId}`);
}
```

**Key points:**
- All mutations are `async` functions marked `'use server'`
- Use `.run()` (synchronous) with better-sqlite3, not `.execute()` (which is async)
- Always call `revalidatePath('/')` to refresh home list
- For entry updates, also call `revalidatePath('/entries/${id}')` to refresh detail page
- Return types: `Promise<{ id: number }>` for create, `Promise<void>` for updates/deletes

### Client Component Autosave Debounce Pattern

**Source:** RESEARCH.md Pattern 5 (Autosave Debounce in Client Component)

**Pattern applies to:** `src/components/editor-form.tsx`

```typescript
'use client';

import { useCallback, useRef, useState } from 'react';

type SaveStatus = 'idle' | 'saving' | 'saved';

export default function EditorForm() {
  const [saveStatus, setSaveStatus] = useState<SaveStatus>('idle');
  const entryIdRef = useRef<number | null>(null);
  const isSavingRef = useRef(false);
  const debounceRef = useRef<ReturnType<typeof setTimeout> | null>(null);
  const fadeTimerRef = useRef<ReturnType<typeof setTimeout> | null>(null);

  const triggerAutosave = useCallback(async (title: string, body: string, tags: string[]) => {
    // Skip if save already in flight (prevents race condition)
    if (isSavingRef.current) return;
    
    // Clear existing debounce timer
    if (debounceRef.current) clearTimeout(debounceRef.current);
    
    debounceRef.current = setTimeout(async () => {
      // Skip if content is empty
      if (!title.trim() && !body.trim()) return;
      
      isSavingRef.current = true;
      setSaveStatus('saving');
      
      try {
        if (entryIdRef.current === null) {
          // First save — create entry and redirect
          const { id } = await createEntry(title, body);
          entryIdRef.current = id;
          await setEntryTags(id, tags);
          router.replace(`/entries/${id}/edit`);
        } else {
          // Subsequent save — update
          await updateEntry(entryIdRef.current, title, body);
          await setEntryTags(entryIdRef.current, tags);
        }
        
        setSaveStatus('saved');
        
        // Fade status after 2500ms
        if (fadeTimerRef.current) clearTimeout(fadeTimerRef.current);
        fadeTimerRef.current = setTimeout(() => setSaveStatus('idle'), 2500);
      } catch (error) {
        console.error('Autosave failed:', error);
        setSaveStatus('idle');
      } finally {
        isSavingRef.current = false;
      }
    }, 500); // 500ms debounce
  }, [router]);

  return (
    // Call triggerAutosave from onChange handlers
    <input
      onChange={(e) => {
        setTitle(e.target.value);
        triggerAutosave(e.target.value, body, chipTags);
      }}
    />
  );
}
```

**Key points:**
- `useRef` for timer handles prevents stale closures and memory leaks
- `isSavingRef` prevents concurrent saves on fast typing
- `entryIdRef` tracks whether DB record exists (null = not created yet)
- `triggerAutosave` is wrapped in `useCallback` with empty dependency array (safe because all state updates via setState)
- 500ms debounce is measured from function call, not from last keystroke
- Fade timer (2500ms) is separate from debounce — allows quick feedback that save succeeded

### MDEditor Dynamic Import with SSR False

**Source:** RESEARCH.md Pattern 1

**Pattern applies to:** `src/components/editor-form.tsx`

```typescript
'use client';

import dynamic from 'next/dynamic';
import '@uiw/react-md-editor/markdown-editor.css';
import '@uiw/react-markdown-preview/markdown.css';

const MDEditor = dynamic(
  () => import('@uiw/react-md-editor'),
  { ssr: false }
);

export default function EditorForm() {
  return (
    <div data-color-mode="light">
      <MDEditor
        value={body}
        onChange={(val) => setBody(val ?? '')}
        height={500}
        preview="live" // or "edit" or "preview"
      />
    </div>
  );
}
```

**Key points:**
- CSS imports MUST be inside the Client Component file, not in `globals.css` (prevents style bleed into detail page)
- `dynamic(() => import(...), { ssr: false })` disables SSR for the MDEditor (it uses browser APIs)
- `data-color-mode="light"` forces light mode regardless of system preference (project scope: light mode only)
- `onChange` handler receives `string | undefined` — use `val ?? ''` to coerce undefined to empty string

### Page Wrapper Pattern (Server Component + Client Child)

**Source:** `src/app/entries/[id]/page.tsx` + `src/components/entry-card.tsx` pattern

**Pattern applies to:** `src/app/new/page.tsx` and `src/app/entries/[id]/edit/page.tsx`

```typescript
// src/app/new/page.tsx (Server Component)
import EditorForm from '@/components/editor-form';

export default function NewEntryPage() {
  return (
    <main className="max-w-5xl mx-auto px-4 py-8">
      <EditorForm mode="new" />
    </main>
  );
}
```

```typescript
// src/app/entries/[id]/edit/page.tsx (Server Component with data fetch)
import { notFound } from 'next/navigation';
import { getEntryById } from '@/lib/actions';
import EditorForm from '@/components/editor-form';

type Props = { params: Promise<{ id: string }> };

export default async function EditEntryPage({ params }: Props) {
  const { id } = await params;
  const numericId = parseInt(id, 10);
  if (isNaN(numericId)) notFound();
  const entry = await getEntryById(numericId);

  if (!entry) return notFound();

  return (
    <main className="max-w-5xl mx-auto px-4 py-8">
      <EditorForm initialEntry={entry} mode="edit" />
    </main>
  );
}
```

**Key points:**
- Page (`page.tsx`) is always a Server Component — no `'use client'`
- Page handles data fetching and error cases (`notFound()`)
- Interactivity lives in Client Component child (`EditorForm`)
- Props flow down: `initialEntry` + `mode` tell EditorForm whether to create or edit
- Layout: `max-w-5xl mx-auto px-4 py-8` (matches all pages for consistency)
